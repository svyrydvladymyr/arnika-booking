arr.sort((a,b)=> {
    let objectFields =  Object.keys(a);
    if(objectFields.length === 0) return 0;
    //TODO: use tolowercase for strings?
    let sortField = field in a ? field : objectFields[0];
    if(a[sortField] < b[sortField]) return -1;
    if(a[sortField] > b[sortField]) return 1;
    return 0;
});
